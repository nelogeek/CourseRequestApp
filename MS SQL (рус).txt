CREATE DATABASE CourseRequest;

CREATE TABLE Roles (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    RoleName NVARCHAR(50) NOT NULL
);

INSERT INTO Roles (RoleName)
VALUES ('Обучаемый'), ('Инициатор'), ('Координатор');

CREATE TABLE Users (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Username NVARCHAR(100) ,
    RoleId INT,
    CONSTRAINT FK_Users_Roles FOREIGN KEY (RoleId) REFERENCES Roles(Id)
);

CREATE TABLE Type (
    id INT IDENTITY(1, 1) PRIMARY KEY,
    type NVARCHAR(50)
);

INSERT INTO Type (type) VALUES
    ('Базовый'),
    ('Продвинутый'),
    ('Для администраторов');


CREATE TABLE Status (
    id INT IDENTITY(1,1) PRIMARY KEY,
    status NVARCHAR(50)
);

INSERT INTO Status (status)
VALUES ('Новая'), ('В работе'), ('Заявлен на обучение'), ('Курс пройден'), ('Курс не пройден') ;


CREATE TABLE Requests (
    id INT IDENTITY(1,1) PRIMARY KEY,
    full_name NVARCHAR(50),
    department NVARCHAR(50),
    position NVARCHAR(50),
    course_name NVARCHAR(300),
    course_type INT,
    notation NVARCHAR(800),
    status INT,
    course_start DATE,
    course_end DATE,
    year INT,
    [user] NVARCHAR(100),
    FOREIGN KEY (course_type) REFERENCES Type(id),
    FOREIGN KEY (status) REFERENCES Status(id)
);
